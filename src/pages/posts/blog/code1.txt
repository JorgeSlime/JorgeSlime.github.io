#include <iostream>
#include <vector>
#include <cstdint>
int main(){
    constexpr int64_t MOD = 1000000007; 
    int n; std::cin >> n; 
    std::vector<int64_t> dp(n + 1,-1);
    auto back=[&](auto&& back,int val)->int{
        if(val == 0) return 1; 
        if(val < 0) return 0; 
        if(dp[val] != -1) return dp[val]; 
        int64_t ans=back(back,val-1) + back(back,val-2) + back(back,val-3); 
        ans %= MOD; 
        return dp[val] = ans; 
    }; 
    std::cout << back(back,n) << '\n'; 
    return 0;
}
